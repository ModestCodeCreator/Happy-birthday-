        // Добавляем звуковой эффект при клике (совместимый с браузерами)
        function playClickSound() {
            try {
                // Простой beep через Web Audio API
                const audioContext = new (window.AudioContext || window.webkitAudioContext)();
                const oscillator = audioContext.createOscillator();
                const gainNode = audioContext.createGain();
                
                oscillator.connect(gainNode);
                gainNode.connect(audioContext.destination);
                
                oscillator.frequency.value = 800;
                gainNode.gain.value = 0.1;
                
                oscillator.start();
                setTimeout(() => {
                    oscillator.stop();
                }, 100);
            } catch (e) {
                // Если Web Audio не поддерживается, ничего не делаем
            }
        }
        
        // Добавляем звук при кликах
        document.querySelectorAll('button, .wish-card').forEach(element => {
            element.addEventListener('click', playClickSound);
        });
